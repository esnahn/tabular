def get_model_trained_xgboost(X_train, X_valid, y_train, y_valid):
    # Set random seed for reproducibility
    seed = 42
    np.random.seed(seed)
    torch.manual_seed(seed)

    model = xgb.XGBRegressor(
        n_estimators=1000,
        learning_rate=0.01,
        early_stopping_rounds=50,
        # max_depth=5,
        random_state=seed,  # Seed for reproducibility
    )
    model.fit(
        X_train,
        y_train,
        eval_set=[
            (X_train, y_train),
            (X_valid, y_valid),
        ],
        verbose=100,
    )
    # Evaluate the model
    r_sq = model.score(X_train, y_train)  # R^2 score on training data
    print(f"R^2 Score on Training Data: {r_sq:.2f}")
    r_sq_valid = model.score(X_valid, y_valid)  # R^2 score on validation data
    print(f"R^2 Score on Validation Data: {r_sq_valid:.2f}")

    return model

def get_model_trained_tabnet(X_train, X_valid, y_train, y_valid):
    # Pandas DataFrame are not supported: apply X.values when calling fit
    X_train = X_train.values
    X_valid = X_valid.values
    # Targets should be 2D : (n_samples, n_regression) but y_train.shape=(n,) given.
    # Use reshape(-1, 1) for single regression.
    y_train = y_train.values.reshape(-1, 1)
    y_valid = y_valid.values.reshape(-1, 1)

    # Set random seed for reproducibility
    seed = 42
    np.random.seed(seed)
    torch.manual_seed(seed)

    model = TabNetRegressor(
        seed=seed,
        verbose=50,  # Print results every verbose iteration
    )

    model.fit(
        X_train,
        y_train,
        eval_set=[
            (X_train, y_train),
            (X_valid, y_valid),
        ],
        eval_name=["train", "valid"],
        eval_metric=["rmse"],
        max_epochs=200,
        patience=10,
    )
    # Evaluate the model
    y_pred_train = model.predict(X_train)
    r_sq = explained_variance_score(y_train, y_pred_train)  # R^2 score on training data
    print(f"R^2 Score on Training Data: {r_sq:.2f}")

    y_pred_valid = model.predict(X_valid)
    r_sq_valid = explained_variance_score(
        y_valid, y_pred_valid
    )  # R^2 score on validation data
    print(f"R^2 Score on Validation Data: {r_sq_valid:.2f}")

    return model

def get_shap_values(model, X_train, X_valid, y_train, y_valid):
    # Set random seed for reproducibility
    seed = 42
    np.random.seed(seed)
    torch.manual_seed(seed)

    # Create SHAP explainer for the model
    # Pandas DataFrame are not supported with TabNet: apply X.values whenever needed
    def f(x):
        # ignore warnings
        # with warnings.catch_warnings(action="ignore"):
        # use scaler to normalize original input value
        # x_norm = scaler.transform(x)

        preds = model.predict(x.values)

        # Dynamically create the slicing tuple
        slicing = (slice(None),) + (0,) * (preds.ndim - 1)
        return preds[slicing]

    explainer = shap.Explainer(f, X_train.iloc[0:100, :])
    shap_values = explainer(X_valid.iloc[0:100, :])
    shap.plots.beeswarm(shap_values)
    plt.show()

    return shap_values
